set -euo pipefail

# ── FLIP-FLOPS DE ESTADO (8 bits = 8 flip-flops D) ───────────────────────
# Cada bit é um flip-flop: SET=1 CLR=0 TOG=XOR
# Abertos linearmente em sequência, fechados em sequência
# Nunca dois bits activos em conflito simultâneo
FL_IDLE=0x00   # 00000000 sem estado
FL_KEY=0x01    # 00000001 verificando API key
FL_NET=0x02    # 00000010 conexão de rede
FL_TX=0x04     # 00000100 enviando dados
FL_RX=0x08     # 00001000 recebendo dados
FL_PARSE=0x10  # 00010000 parsing resposta
FL_FILE=0x20   # 00100000 operação de ficheiro
FL_ERR=0x40    # 01000000 estado de erro
FL_DONE=0x80   # 10000000 concluído

_F=0  # registrador de 8 flip-flops

# Operações branchless sobre flip-flops
FF_SET(){ _F=$((_F|$1)); }
FF_CLR(){ _F=$((_F&~($1))); }
FF_GET(){ ((_F&$1)); }
FF_TOG(){ _F=$((_F^$1)); }

# ── CORES ANSI ────────────────────────────────────────────────────────────
R='\033[0;31m' G='\033[0;32m' Y='\033[1;33m' B='\033[0;34m'
M='\033[0;35m' C='\033[0;36m' W='\033[1;37m' Z='\033[0m'
BLD='\033[1m'  DIM='\033[2m'  UL='\033[4m'

# ── I/O LINEAR (sem cursor move, sem clear, sem rewind) ───────────────────
# Saída flui para a frente, nunca para trás — absorção de fricção linear
P(){ printf "%b" "$*"; }
PL(){ printf "%b\n" "$*"; }
SEP(){ PL "${C}${BLD}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${Z}"; }
HDR(){ PL "\n${M}${BLD}▸ $* ${Z}"; }

# ── DETECÇÃO DE ARQUITECTURA ──────────────────────────────────────────────
_ARCH=$(uname -m)
case "$_ARCH" in
  aarch64|arm64)  _A="ARM64" ;;
  armv7*|armv8l*) _A="ARM32" ;;
  x86_64)         _A="x86-64" ;;
  *)              _A="GENERIC" ;;
esac

# ── JSON ESCAPE (pure bash — sem sed, sem python, sem awk) ───────────────
# Máquina de estados linear: varre string byte a byte, emite escape
# Sem heap: opera sobre stack bash (local vars)
_JESC(){
  local _s="$1" _o="" _i _c
  for((_i=0;_i<${#_s};_i++)); do
    _c="${_s:$_i:1}"
    case "$_c" in
      '"')   _o="${_o}\\\"" ;;
      $'\\') _o="${_o}\\\\" ;;
      $'\n') _o="${_o}\\n"  ;;
      $'\r') _o="${_o}\\r"  ;;
      $'\t') _o="${_o}\\t"  ;;
      $'\b') _o="${_o}\\b"  ;;
      $'\f') _o="${_o}\\f"  ;;
      *)     _o="${_o}${_c}" ;;
    esac
  done
  printf '%s' "$_o"
}

# ── JSON PARSE: extrai "text" da resposta Anthropic (pure bash) ───────────
# Flip-flop de 3 estados: SEEK → IN_KEY → IN_VAL
# Linear: avança byte a byte, nunca recua
_JGET_TEXT(){
  local _r="$1"
  # Localiza o marcador "text":"
  local _m='"text":"'
  local _tail="${_r#*$_m}"
  # Se não encontrou, tenta array content
  if [ "${#_tail}" -eq "${#_r}" ]; then
    _m='"text": "'
    _tail="${_r#*$_m}"
  fi
  [ "${#_tail}" -eq "${#_r}" ] && { printf '%s' "[resposta vazia ou erro de parse]"; return; }
  # Extrai até próximo " não-escaped
  local _o="" _i _c _p=""
  for((_i=0;_i<${#_tail};_i++)); do
    _c="${_tail:$_i:1}"
    if [ "$_p" = $'\\' ]; then
      case "$_c" in
        n)   _o="${_o}"$'\n' ;;
        r)   _o="${_o}"$'\r' ;;
        t)   _o="${_o}"$'\t' ;;
        '"') _o="${_o}\""    ;;
        $'\\') _o="${_o}\\"  ;;
        *)   _o="${_o}${_c}" ;;
      esac
      _p=""
    elif [ "$_c" = $'\\' ]; then
      _p=$'\\'
    elif [ "$_c" = '"' ]; then
      break
    else
      _o="${_o}${_c}"
      _p=""
    fi
  done
  printf '%s' "$_o"
}

# ── JSON PARSE: extrai tokens de uso ─────────────────────────────────────
_JGET_USAGE(){
  local _r="$1"
  local _in="${_r#*\"input_tokens\":}"  ; _in="${_in%%,*}" ; _in="${_in%%}*}"
  local _out="${_r#*\"output_tokens\":}"; _out="${_out%%,*}"; _out="${_out%%}*}"
  printf 'in=%s out=%s' "${_in// /}" "${_out// /}"
}

# ── JSON PARSE: extrai status code HTTP ──────────────────────────────────
_HTTP_STATUS(){
  local _r="$1"
  local _l="${_r%%$'\r'*}"
  local _c="${_l#HTTP/* }"
  printf '%s' "${_c%% *}"
}

# ── HTTP BODY: localiza após \r\n\r\n ─────────────────────────────────────
_HTTP_BODY(){
  local _r="$1"
  printf '%s' "${_r#*$'\r\n\r\n'}"
}

# ── UNCHUNK: remove Transfer-Encoding: chunked (pure bash) ───────────────
# Máquina de estados linear: lê tamanho hex, consome N bytes, avança
_UNCHUNK(){
  local _r="$1" _out="" _line _sz _chunk
  while [ -n "$_r" ]; do
    _line="${_r%%$'\r\n'*}"
    _r="${_r#*$'\r\n'}"
    # Tamanho em hex (ignora extensões de chunk após ';')
    _line="${_line%%;*}"
    _line="${_line// /}"
    [ -z "$_line" ] && continue
    _sz=$(( 16#${_line} )) 2>/dev/null || break
    [ $_sz -eq 0 ] && break
    _chunk="${_r:0:$_sz}"
    _out="${_out}${_chunk}"
    _r="${_r:$((_sz+2))}"
  done
  printf '%s' "$_out"
}

# ── TLS POST via openssl s_client (sem curl, sem wget) ───────────────────
# Pipe linear: request → openssl TLS → Anthropic → response
# Freestanding: openssl é o único binário externo necessário
# Absorção de fricção: retry automático em falha de rede
_API_SEND(){
  local _body="$1" _blen=${#1}
  FF_SET $FL_TX

  # Constrói HTTP/1.1 request raw — sem sprintf, byte a byte
  local _req
  printf -v _req \
    "POST /v1/messages HTTP/1.1\r\nHost: api.anthropic.com\r\nContent-Type: application/json\r\nx-api-key: %s\r\nanthropic-version: 2023-06-01\r\nContent-Length: %d\r\nConnection: close\r\n\r\n%s" \
    "$ANTHROPIC_API_KEY" "$_blen" "$_body"

  FF_CLR $FL_TX
  FF_SET $FL_RX

  # Pipe raw: printf → openssl TLS → captura resposta completa
  local _resp
  _resp=$(printf '%s' "$_req" | \
    openssl s_client \
      -connect api.anthropic.com:443 \
      -servername api.anthropic.com \
      -quiet \
      -no_ign_eof \
      2>/dev/null) || true

  FF_CLR $FL_RX
  printf '%s' "$_resp"
}

# ── HISTÓRIA DE CONVERSA (linear append-only, sem rewind) ─────────────────
# _H = JSON array de {role, content} que só cresce para a frente
# Determinismo matemático: cada turno é imutável após inserção
_H='[]'
_HC=0

_HIST_ADD(){
  local _ro="$1" _co
  _co="$(_JESC "$2")"
  if [ "$_H" = '[]' ]; then
    _H='[{"role":"'"$_ro"'","content":"'"$_co"'"}]'
  else
    _H="${_H%]}"
    _H="${_H}"',{"role":"'"$_ro"'","content":"'"$_co"'"}'
    _H="${_H}]"
  fi
  _HC=$((_HC+1))
}

_HIST_RESET(){
  _H='[]'; _HC=0
  PL "${Y}[RESET] Contexto limpo — nova conversa linear${Z}"
}

_HIST_SIZE(){
  printf '%d turnos · %dB histórico' "$_HC" "${#_H}"
}

# ── SYSTEM PROMPT (injectado na 1ª chamada) ───────────────────────────────
_SYS='Você é GaiaPhi, uma IA freestanding sem abstração. Responda de forma directa, densa e matemática. Absorva fricção. GaiaPhi=Omega=Amor.'

# ── CHAMADA PRINCIPAL À IA ────────────────────────────────────────────────
_AI_CALL(){
  local _q="$1"
  _HIST_ADD "user" "$_q"

  # Constrói body JSON com sistema + histórico
  local _body
  _body='{"model":"claude-sonnet-4-20250514","max_tokens":2048,"system":"'"$(_JESC "$_SYS")"'","messages":'"$_H"'}'

  PL "${DIM}${C}[TX] ${#_body}B → api.anthropic.com · flags=$(printf '0x%02X' $_F)${Z}"

  # Absorção de fricção: tenta até 3 vezes
  local _raw _status _body_json _text _usage _try=0
  while [ $_try -lt 3 ]; do
    _try=$((_try+1))
    _raw=$(_API_SEND "$_body")
    _status=$(_HTTP_STATUS "$_raw")
    [ "$_status" = "200" ] && break
    PL "${Y}[RETRY $_try/3] HTTP $_status${Z}"
    sleep 1
  done

  _body_json=$(_HTTP_BODY "$_raw")

  # Detecta chunked e remove encoding
  case "$_raw" in
    *"Transfer-Encoding: chunked"*)
      _body_json=$(_UNCHUNK "$_body_json") ;;
  esac

  PL "${DIM}${C}[RX] HTTP $_status · ${#_body_json}B${Z}"

  if [ "$_status" != "200" ]; then
    FF_SET $FL_ERR
    PL "${R}[ERR] HTTP $_status${Z}"
    PL "${DIM}${_body_json:0:300}${Z}"
    FF_CLR $FL_ERR
    return 1
  fi

  FF_SET $FL_PARSE
  _text=$(_JGET_TEXT "$_body_json")
  _usage=$(_JGET_USAGE "$_body_json")
  FF_CLR $FL_PARSE

  # Adiciona resposta ao histórico linear — imutável após inserção
  _HIST_ADD "assistant" "$_text"

  printf '%s' "$_text"
  PL "\n\n${DIM}${M}[${_usage} · $(_HIST_SIZE) · flags=$(printf '0x%02X' $_F)]${Z}"
}

# ── FILE GET: lê ficheiro e injeta no contexto ────────────────────────────
# Sem malloc: ficheiro lido para variável bash (string na stack)
_FILE_GET(){
  local _f="$1"
  FF_SET $FL_FILE
  if [ ! -f "$_f" ]; then
    PL "${R}[GET ERR] Ficheiro não encontrado: $_f${Z}"
    FF_CLR $FL_FILE; return 1
  fi
  local _c _sz
  _c=$(cat "$_f")
  _sz=${#_c}
  PL "${G}[GET] $_f · ${_sz}B → injectado no contexto${Z}"
  _HIST_ADD "user" "Ficheiro recebido: $_f"$'\n'"$(printf '```\n%s\n```' "$_c")"
  FF_CLR $FL_FILE
}

# ── FILE POST: guarda resposta num ficheiro ───────────────────────────────
# Linear write: abre, escreve para a frente, fecha — sem seek, sem rewind
_FILE_POST(){
  local _f="$1" _c="$2"
  FF_SET $FL_FILE
  printf '%s\n' "$_c" > "$_f" && \
    PL "${G}[POST] Guardado → $_f · ${#_c}B${Z}" || \
    PL "${R}[POST ERR] Falha ao guardar: $_f${Z}"
  FF_CLR $FL_FILE
}

# ── VERIFICAÇÃO DE DEPENDÊNCIAS ───────────────────────────────────────────
_CHK(){
  FF_SET $FL_KEY
  command -v openssl >/dev/null 2>&1 || {
    PL "${R}[ERR] openssl não encontrado — obrigatório para TLS${Z}"
    PL "${Y}  Termux : pkg install openssl${Z}"
    PL "${Y}  Debian : apt install openssl${Z}"
    PL "${Y}  Alpine : apk add openssl${Z}"
    FF_SET $FL_ERR; exit 1
  }
  PL "${G}[OK] openssl $(openssl version | cut -d' ' -f2)${Z}"
  if [ -z "${ANTHROPIC_API_KEY:-}" ]; then
    P "${Y}[KEY] ANTHROPIC_API_KEY não definida. Cola aqui: ${Z}"
    IFS= read -rs ANTHROPIC_API_KEY
    printf '\n'
    export ANTHROPIC_API_KEY
  fi
  [ -z "${ANTHROPIC_API_KEY:-}" ] && {
    PL "${R}[ERR] API key vazia — define: export ANTHROPIC_API_KEY=sk-ant-...${Z}"
    FF_SET $FL_ERR; exit 1
  }
  PL "${G}[OK] API key: ${ANTHROPIC_API_KEY:0:12}...${Z}"
  FF_CLR $FL_KEY
}

# ── PARSER DE COMANDOS INTERNOS ───────────────────────────────────────────
# Flip-flop de decisão: 0=vai à IA  2=comando interno tratado
_LAST_RESP=""

_CMD(){
  local _i="$1" _cmd _arg
  _cmd="${_i%% *}"
  _arg="${_i#* }"
  [ "$_arg" = "$_i" ] && _arg=""

  case "$_cmd" in
    /get)
      [ -z "$_arg" ] && { PL "${Y}Uso: /get <ficheiro>${Z}"; return 2; }
      _FILE_GET "$_arg"; return 2 ;;
    /post)
      [ -z "$_arg" ] && { PL "${Y}Uso: /post <ficheiro>${Z}"; return 2; }
      [ -z "$_LAST_RESP" ] && { PL "${Y}[POST] Sem resposta para guardar ainda${Z}"; return 2; }
      _FILE_POST "$_arg" "$_LAST_RESP"; return 2 ;;
    /reset)
      _HIST_RESET; _LAST_RESP=""; return 2 ;;
    /flags)
      HDR "Estado dos Flip-Flops"
      printf "${M}"
      printf "  Registrador : 0x%02X\n" $_F
      printf "  KEY    = %d\n" $(((_F&FL_KEY)!=0))
      printf "  NET    = %d\n" $(((_F&FL_NET)!=0))
      printf "  TX     = %d\n" $(((_F&FL_TX)!=0))
      printf "  RX     = %d\n" $(((_F&FL_RX)!=0))
      printf "  PARSE  = %d\n" $(((_F&FL_PARSE)!=0))
      printf "  FILE   = %d\n" $(((_F&FL_FILE)!=0))
      printf "  ERR    = %d\n" $(((_F&FL_ERR)!=0))
      printf "  DONE   = %d\n" $(((_F&FL_DONE)!=0))
      printf "${Z}"
      return 2 ;;
    /hist)
      PL "${M}$(_HIST_SIZE)${Z}"; return 2 ;;
    /sys)
      [ -n "$_arg" ] && { _SYS="$_arg"; PL "${G}[SYS] System prompt actualizado${Z}"; } || \
        PL "${M}System: $_SYS${Z}"
      return 2 ;;
    /raw)
      # Mostra última resposta HTTP raw
      PL "${DIM}${_LAST_RESP:0:500}${Z}"; return 2 ;;
    /help|/?)
      SEP
      PL "${W}${BLD}GaiaPhi — Comandos${Z}"
      SEP
      PL "  ${C}/get  <f>${Z}   → lê ficheiro e injeta no contexto (POST de ficheiro para IA)"
      PL "  ${C}/post <f>${Z}   → guarda última resposta em ficheiro (GET de resposta da IA)"
      PL "  ${C}/reset${Z}      → limpa histórico de conversa"
      PL "  ${C}/flags${Z}      → mostra estado dos 8 flip-flops"
      PL "  ${C}/hist${Z}       → contagem de turnos e tamanho do histórico"
      PL "  ${C}/sys  [txt]${Z} → lê ou define system prompt"
      PL "  ${C}/raw${Z}        → mostra última resposta crua"
      PL "  ${C}/exit${Z}       → termina"
      PL "  ${Y}<texto>${Z}     → envia mensagem à IA"
      SEP
      return 2 ;;
    /exit|/quit|/q|/sair)
      FF_SET $FL_DONE
      PL "${G}GaiaPhi · Ω=Amor · $(_HIST_SIZE) · saindo${Z}"
      exit 0 ;;
  esac
  return 0  # 0 = pergunta normal para a IA
}

# ── PROMPT VISUAL ─────────────────────────────────────────────────────────
_PROMPT(){
  printf "\n${BLD}${G}GΦ${Z}${W}[%d]${Z}${Y}❯${Z} " "$_HC"
}

# ── REPL PRINCIPAL (linear, forward-only, sem goto, sem recursão) ─────────
# Cada iteração do while avança o estado — nunca recua
# Absorção de fricção: erros de rede não terminam o loop
_REPL(){
  local _inp _rc

  while true; do
    _PROMPT
    IFS= read -r _inp || { printf '\n'; break; }
    [ -z "${_inp// /}" ] && continue

    _CMD "$_inp"
    _rc=$?
    [ $_rc -eq 2 ] && continue  # comando interno: próxima iteração

    # Pergunta à IA — renderização linear para a frente
    SEP
    PL "${DIM}${C}[IA] processando...${Z}"
    SEP

    local _resp
    _resp=$(_AI_CALL "$_inp") && {
      _LAST_RESP="$_resp"
      SEP
      PL "${W}${BLD}${_resp}${Z}"
      SEP
    } || {
      PL "${R}[ERR] Falha na chamada à IA — histórico preservado${Z}"
    }
  done
}

# ── MODO NÃO-INTERACTIVO: recebe pergunta como argumento ──────────────────
_ONESHOT(){
  local _q="$1"
  PL "${DIM}${C}[ONESHOT] $_q${Z}"
  SEP
  local _r
  _r=$(_AI_CALL "$_q") && PL "${W}${_r}${Z}" || PL "${R}[ERR]${Z}"
  SEP
}

# ── ENTRY POINT ───────────────────────────────────────────────────────────
_MAIN(){
  _LOGO
  SEP
  PL "${M}Arch: $_A ($(uname -m)) · Bash: $BASH_VERSION${Z}"
  PL "${M}Modelo: claude-sonnet-4-20250514 · Anthropic v1/messages${Z}"
  PL "${M}Modo: freestanding · TLS pipe openssl · HTTP/1.1 raw${Z}"
  PL "${M}GET/POST ficheiros · histórico linear · absorção de fricção${Z}"
  SEP

  _CHK

  PL "${G}[OK] GaiaPhi iniciado · /help para comandos${Z}"
  PL "${DIM}GaiaPhi=Ω · DeltaRafaelVerboOmega · F*=23.158 · freestanding${Z}"
  SEP

  # Modo oneshot: se há argumentos, responde e sai
  if [ $# -gt 0 ]; then
    _ONESHOT "$*"
    FF_SET $FL_DONE
    exit 0
  fi

  # Modo interactivo: REPL linear
  _REPL
}

_MAIN "$@"
